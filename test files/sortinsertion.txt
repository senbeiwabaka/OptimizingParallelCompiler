var
   int i
   int j
   int t
   int x
   int maxnum
   int seed
   int lb
   int ub
   list[30] data
begin
   prompt "Please input a non-zero random number generator seed: "
   let seed = -1
   while (seed < 0)
      input seed
      if (seed >= 0) then goto break
      prompt "I need a non-zero number: "
      let seed = -1
   endwhile
   label break
   let maxnum = 30
   let lb = 0
   let ub = maxnum - 1
   rem fill
   for i = lb to ub
      let seed = (65 * seed) % 1021
      let data[i] = seed
   endfor
   rem end fill
   rem insert sort
   for i = lb + 1 to ub
      let t = data[i]
      for j = lb to i - 1
         let x = i - 1 - j
         if (data[x + lb] > t) then let data[x + lb + 1] = data[x + lb]
      endfor
      let data[j + 1] = t
   endfor
   rem end insert sort
   prompt "results:\n"
   for i = lb to ub
      print i
      prompt ": "
      print data[i]
      prompt "\n"
   endfor
end
